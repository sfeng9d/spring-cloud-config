3   import lombok.Setter;
4   import lombok.extern.slf4j.Slf4j;
5   
6   import org.springframework.beans.factory.annotation.Autowired;
7   import org.springframework.security.core.Authentication;
8   import org.springframework.security.core.context.SecurityContextHolder;
9   
10  /**
11   * Determine whether let user go further by login user and repository name in URL.
12   */
13  @Slf4j
14  public class ApplicationConfigSecurity {
15  
16    @Autowired
17    @Setter
18    private ConfigServerSecurityProperties configServerSecurityProperties;
19  
20    /**
21     * Allow admin user, allow user that name matches repository name
22     *
23     * @param pathRepositoryName repository name extracted from path
24     * @return allow access
25     */
26    public boolean checkAuthenticationName(final String pathRepositoryName) {
27      final boolean result;
28  
29      final Authentication authentication = SecurityContextHolder.getContext().getAuthentication();
30      final String authenticationName = authentication != null ? authentication.getName() : null;
31      if (authenticationName != null) {
32        if (this.configServerSecurityProperties.getAdminName().equals(authenticationName) && authentication.isAuthenticated()) {
33          result = true;
34        } else {
35          result = pathRepositoryName != null && pathRepositoryName.equals(authenticationName);
36        }
37      } else {
38        result = false;
39      }
40  
41      log.debug("'{}' {} privilege to access '{}'", authenticationName, result ? "has" : "has no", pathRepositoryName);
42  
43      return result;
44    }
45  }
