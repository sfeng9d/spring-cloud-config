3   import static org.apache.commons.lang3.StringUtils.isBlank;
4   
5   import lombok.Getter;
6   import lombok.Setter;
7   import lombok.extern.slf4j.Slf4j;
8   
9   import org.springframework.beans.factory.annotation.Value;
10  
11  import java.util.UUID;
12  
13  import javax.annotation.PostConstruct;
14  
15  @Getter
16  @Setter
17  @Slf4j
18  public class ConfigServerSecurityProperties {
19  
20    @Value("${security.user.name:admin}")
21    private String adminName;
22  
23    @Value("${security.user.password:}")
24    private String adminPassword;
25  
26    @Value("${security.hook.name:hook}")
27    private String hookName;
28  
29    @Value("${security.hook.password:hook_pass}")
30    private String hookPassword;
31  
32    @PostConstruct
33    private void init() {
34      if (isBlank(this.adminPassword)) {
35        this.adminPassword = UUID.randomUUID().toString();
36        log.info("auto generated admin password, username:{}, password:{}", this.adminName, this.adminPassword);
37      }
38  
39      if (isBlank(this.hookPassword)) {
40        this.hookPassword = UUID.randomUUID().toString();
41        log.info("auto generated hook password, username:{}, password:{}", this.hookName, this.hookPassword);
42      }
43    }
44  }
