1   package cn.home1.cloud.config.server.ssh;
2   
3   import static org.springframework.util.FileCopyUtils.copyToString;
4   
5   import cn.home1.cloud.config.server.util.Consts;
6   import cn.home1.cloud.config.server.util.ResourceUtils;
7   
8   import lombok.SneakyThrows;
9   
10  import org.springframework.beans.factory.annotation.Autowired;
11  import org.springframework.cloud.config.server.ssh.SshUriProperties;
12  
13  import java.io.File;
14  import java.io.FileReader;
15  
16  import javax.annotation.PostConstruct;
17  
18  public class DeployKey {
19  
20    private final String privateKeyLocation;
21  
22    @Autowired
23    private SshUriProperties sshUriProperties;
24  
25    public DeployKey(final String privateKeyLocation) {
26      this.privateKeyLocation = privateKeyLocation;
27    }
28  
29  
30    @PostConstruct
31    public void init() {
32      final String privateKey = this.getPrivateKey();
33  
34      // setup deploy key
35      //SshSessionFactory.setInstance(new CustomJschConfigSessionFactory(file));
36      this.sshUriProperties.setPrivateKey(privateKey);
37      this.sshUriProperties.setIgnoreLocalSshSettings(true);
38    }
39  
40    public String getPrivateKeyPath() {
41      return ResourceUtils.findResourceFile(this.privateKeyLocation, Consts.DATA_DIRECTORY);
42    }
43  
44    @SneakyThrows
45    public String getPrivateKey() {
46      return copyToString(new FileReader(new File(this.getPrivateKeyPath())));
47    }
48  
49    public String getPublicKeyPath() {
50      return this.getPrivateKeyPath() + ".pub";
51    }
52  
53    @SneakyThrows
54    public String getPublicKey() {
55      return copyToString(new FileReader(new File(this.getPublicKeyPath())));
56    }
57  }
